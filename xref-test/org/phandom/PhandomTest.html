32  import com.rexsl.test.XhtmlMatchers;
33  import org.apache.commons.io.IOUtils;
34  import org.apache.commons.lang3.StringUtils;
35  import org.hamcrest.MatcherAssert;
36  import org.junit.Assume;
37  import org.junit.Before;
38  import org.junit.Test;

40  /**
41   * Test case for {@link Phandom}.
42   * @author Yegor Bugayenko (yegor@tpc2.com)
43   * @version $Id$
44   */
45  public final class PhandomTest {
46  
47      /**
48       * Check for Phantomjs availability.
49       */
50      @Before
51      public void installed() {
52          Assume.assumeTrue(Phandom.isInstalled());
53      }
54  
55      /**
56       * Phandom can build DOM document.
57       * @throws Exception If some problem inside
58       */
59      @Test
60      public void buildsDomDocument() throws Exception {
61          MatcherAssert.assertThat(
62              XhtmlMatchers.xhtml(
63                  new Phandom(
64                      StringUtils.join(
65                          "<!DOCTYPE html>",
66                          "<html><head>",
67                          "<meta content='hi there' name='description'/>",
68                          "</head><body><p>&euro;</p><a href='#'/></body></html>"
69                      )
70                  ).dom()
71              ),
72              XhtmlMatchers.hasXPaths(
73                  "/html/body",
74                  "/html/head/meta[@name='description']",
75                  "//p[.='\u20ac']"
76              )
77          );
78      }
79  
80      /**
81       * Phandom can succeed on a broken DOM.
82       * @throws Exception If some problem inside
83       */
84      @Test
85      public void succeedsOnBrokenDom() throws Exception {
86          MatcherAssert.assertThat(
87              XhtmlMatchers.xhtml(
88                  new Phandom(IOUtils.toInputStream("<html> broken")).dom()
89              ),
90              XhtmlMatchers.hasXPath("/html[head and body]")
91          );
92      }
93  
94      /**
95       * Phandom can fail on a broken javascript.
96       * @throws Exception If some problem inside
97       */
98      @Test(expected = RuntimeException.class)
99      public void failsOnBrokenJavascript() throws Exception {
100         new Phandom(
101             "<html><body><script>a.call();</script></body></html>"
102         ).dom();
103     }

105 }
