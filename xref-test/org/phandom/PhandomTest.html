43      /**
44       * Phandom can build DOM document.
45       * @throws Exception If some problem inside
46       */
47      @Test
48      public void buildsDomDocument() throws Exception {
49          MatcherAssert.assertThat(
50              new Phandom("<html><body><p>Hey!</p></body></html>").dom(),
51              XhtmlMatchers.hasXPaths(
52                  "/xhtml:html/xhtml:body",
53                  "//xhtml:p[.='Hey!']"
54              )
55          );
56      }
