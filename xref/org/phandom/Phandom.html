52  /**
53   * PhantomJS DOM.
54   *
55   * @author Yegor Bugayenko (yegor@tpc2.com)
56   * @version $Id$
57   * @since 0.1
58   */
59  @Immutable
60  @ToString
61  @EqualsAndHashCode(of = "page")
62  @Loggable(Loggable.DEBUG)
63  public final class Phandom {
64  
65      /**
66       * Content of the page to render.
67       */
68      private final transient String page;
69  
70      /**
71       * Public ctor.
72       * @param content Content to encapsulate
73       */
74      public Phandom(@NotNull final String content) {
75          this.page = content;
76      }
77  
78      /**
79       * Public ctor.
80       * @param stream Stream with content
81       * @throws IOException If fails to read the stream
82       */
83      public Phandom(@NotNull final InputStream stream) throws IOException {
84          this(IOUtils.toString(stream, CharEncoding.UTF_8));
85      }
86  
87      /**
88       * Get DOM.
89       * @return DOM
90       * @throws IOException If fails
91       */
92      public Document dom() throws IOException {
93          final Process proc = this.builder().start();
94          proc.getOutputStream().close();
95          return Phandom.parse(
96              new VerboseProcess(proc, Level.FINE, Level.FINE).stdout()
97          );
98      }
99  
100     /**
101      * Create process builder.
102      * @return Builder
103      * @throws IOException If fails
104      */
105     public ProcessBuilder builder() throws IOException {
106         final File script = Phandom.temp(
107             this.getClass().getResourceAsStream("dom.js"),
108             ".js"
109         );
110         final File src = Phandom.temp(
111             new ByteArrayInputStream(this.page.getBytes(Charsets.UTF_8)),
112             ".html"
113         );
114         return new ProcessBuilder(
115             "phantomjs", script.getAbsolutePath(), src.getAbsolutePath()
116         );
117     }
118 
119     /**
120      * Parse XML into DOM.
121      * @param xml XML to parse
122      * @return DOM
123      * @throws IOException If fails
124      */
125     private static Document parse(final String xml) throws IOException {
126         try {
127             return DocumentBuilderFactory.newInstance()
128                 .newDocumentBuilder()
129                 .parse(IOUtils.toInputStream(xml, CharEncoding.UTF_8));
130         } catch (ParserConfigurationException ex) {
131             throw new IOException(xml, ex);
132         } catch (SAXException ex) {
133             throw new IOException(xml, ex);
134         }
135     }
136 
137     /**
138      * Make temporary file.
139      * @param content Content to save there
140      * @param ext Extension
141      * @return File name
142      * @throws IOException If fails
143      */
144     private static File temp(final InputStream content, final String ext)
145         throws IOException {
146         final File file = File.createTempFile("phandom-", ext);
147         IOUtils.copy(content, new FileOutputStream(file));
148         return file;
149     }
