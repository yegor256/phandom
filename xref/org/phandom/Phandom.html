54  /**
55   * PhantomJS DOM.
56   *
57   * <p>Use it to parse XML/XHTML/HTML document using PhantomJS, for example:
58   *
59   * <pre>Document dom = new Phandom(
60   *   "&lt;html&gt;&lt;p&gt;Hey!&lt;/p&gt;&lt;/html&gt;"
61   * ).dom();
62   * Element element = dom.getElementByTag("p");</pre>
63   *
64   * <p>The most popular use case for the class would be its usage
65   * in a unit test, to make sure your HTML document (together with its
66   * embedded JavaScript scripts) is renderable by a browser:
67   *
68   * <pre>import com.rexsl.test.XhtmlMatchers;
69   * import org.hamcrest.MatcherAssert;
70   * import org.junit.Assume;
71   * import org.junit.Test;
72   * import org.phandom.Phandom;
73   * public class HtmlTest {
74   *   &#64;Test
75   *   public void testPageRenderability() {
76   *     Assume.assumeTrue(Phandom.isInstalled());
77   *     MatcherAssert.assertThat(
78   *       new Phandom("&lt;html>&lt;p>Hey!&lt;/p>&lt;/html>").dom(),
79   *       XhtmlMatchers.hasXPath("//body/p[.='Hey!']")
80   *     );
81   *   }
82   * }</pre>
83   *
84   * @author Yegor Bugayenko (yegor@tpc2.com)
85   * @version $Id$
86   * @since 0.1
87   */
88  @Immutable
89  @ToString
90  @EqualsAndHashCode(of = "page")
91  @Loggable(Loggable.DEBUG)
92  public final class Phandom {
93  
94      /**
95       * Name of binary phantomjs.
96       */
97      private static final String BIN = "phantomjs";
98  
99      /**
100      * Pattern to match phantomjs version.
101      */
102     private static final Pattern VERSION =
103         Pattern.compile("\\d+\\.\\d+\\.\\d+");
104 
105     /**
106      * Content of the page to render.
107      */
108     private final transient String page;
109 
110     /**
111      * Public ctor.
112      * @param content Content to encapsulate
113      */
114     public Phandom(@NotNull final String content) {
115         this.page = content;
116     }
117 
118     /**
119      * Public ctor.
120      * @param stream Stream with content
121      * @throws IOException If fails to read the stream
122      */
123     public Phandom(@NotNull final InputStream stream) throws IOException {
124         this(IOUtils.toString(stream, CharEncoding.UTF_8));
125     }
126 
127     /**
128      * PhantomJS binary is installed?
129      * @return TRUE if installed
130      * @since 0.2
131      */
132     public static boolean isInstalled() {
133         boolean installed;
134         try {
135             installed = Phandom.VERSION.matcher(
136                 new VerboseProcess(
137                     new ProcessBuilder(Phandom.BIN, "--version")
138                 ).stdoutQuietly()
139             ).matches();
140         } catch (IllegalStateException ex) {
141             installed = false;
142         }
143         return installed;
144     }
145 
146     /**
147      * Get DOM.
148      * @return DOM
149      * @throws IOException If fails
150      */
151     public Document dom() throws IOException {
152         final Process process = this.builder().start();
153         process.getOutputStream().close();
154         return Phandom.parse(
155             new VerboseProcess(process, Level.FINE, Level.FINE).stdout()
156         );
157     }
158 
159     /**
160      * Create process builder.
161      * @return Builder
162      * @throws IOException If fails
163      */
164     public ProcessBuilder builder() throws IOException {
165         final InputStream src = this.getClass().getResourceAsStream("dom.js");
166         try {
167             return new ProcessBuilder(
168                 Phandom.BIN,
169                 Phandom.temp(src, ".js").getAbsolutePath(),
170                 Phandom.temp(
171                     new ByteArrayInputStream(
172                         this.page.getBytes(Charsets.UTF_8)
173                     ),
174                     ".html"
175                 ).getAbsolutePath()
176             );
177         } finally {
178             src.close();
179         }
180     }
181 
182     /**
183      * Parse XML into DOM.
184      * @param xml XML to parse
185      * @return DOM
186      * @throws IOException If fails
187      */
188     private static Document parse(final String xml) throws IOException {
189         try {
190             return DocumentBuilderFactory.newInstance()
191                 .newDocumentBuilder()
192                 .parse(IOUtils.toInputStream(xml, CharEncoding.UTF_8));
193         } catch (ParserConfigurationException ex) {
194             throw new IOException(xml, ex);
195         } catch (SAXException ex) {
196             throw new IOException(xml, ex);
197         }
198     }
199 
200     /**
201      * Make temporary file.
202      * @param content Content to save there
203      * @param ext Extension
204      * @return File name
205      * @throws IOException If fails
206      */
207     private static File temp(final InputStream content, final String ext)
208         throws IOException {
209         final File file = File.createTempFile("phandom-", ext);
210         final OutputStream output = new FileOutputStream(file);
211         try {
212             IOUtils.copy(content, output);
213         } finally {
214             output.close();
215         }
216         return file;
217     }
